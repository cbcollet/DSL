# Summarize means/SDs
overall_summary <- divorce_data %>%
  select(1:54) %>%
  summarise(across(.cols = everything(),
                   .fns = list(mean = ~mean(., na.rm = TRUE),

# Pivot with dot separator

  pivot_longer(everything(), names_to = c("Variable", ".value"), names_sep = "\\.") %>%
